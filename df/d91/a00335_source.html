53 void ExecdSocket::execute(Enums::NetworkCommand command) noexcept
54 {
55 QByteArray com;
56
57 switch(command)
58 {
59 case Enums::FormImage:
60 com = wrap(Config::get()->value<QString>("EXECD_FORM_TELESCOPIC") + m_args->formArgumentString());
61 break;
62 case Enums::FocusImage:
63 com = wrap(Config::get()->value<QString>("EXECD_FOCUS_TELESCOPIC") + m_args->focusArgumentString());
64 break;
65 case Enums::ReformImage:
66 com = wrap(Config::get()->value<QString>("EXECD_FORM_TELESCOPIC") + m_args->reformArgumentString());
67 break;
68 case Enums::RemoteStorageStatus:
69 com = wrap(Config::get()->value<QString>("EXECD_STORAGE_FETCH"));
70 break;
71 case Enums::ClearRemoteStorage:
72 com = wrap(Config::get()->value<QString>("EXECD_STORAGE_CLEAR") + " " + Config::get()->value<QString>("EXECD_STORAGE_FETCH"));
73 break;
74 case Enums::SimpleStrip:
75 com = wrap(Config::get()->value<QString>("EXECD_FORM_STRIP_DEBUG") + m_args->formArgumentString());
76 break;
77 case Enums::StartStrip:
78 com = wrap(Config::get()->value<QString>("EXECD_FORM_STRIP_START") + m_args->formArgumentString());
79 break;
80 case Enums::StopStrip:
81 com = wrap(Config::get()->value<QString>("EXECD_FORM_STRIP_STOP") + m_args->formArgumentString());
82 break;
83 case Enums::Reboot:
84 {
85 com = wrap(Config::get()->value<QString>("EXECD_REBOOT"));
86 break;
87 }
88 case Enums::PowerOff:
89 {
90 com = wrap(Config::get()->value<QString>("EXECD_POWEROFF"));

95 return;

98 this->send(com);

100 qDebug().noquote() << "[EXECD] Sended built-in command";

106 void ExecdSocket::process(QByteArray data)
107 {
108 QString raw = data.data();
109 QString check_crc = raw;
110 check_crc.chop(5);
111
112 uint16_t crc16 = Utils::crc16(Utils::str_data(check_crc), check_crc.length());
113 uint16_t receivedCrc16 = raw.split("|").last().toUInt(nullptr, 16);
114
115 if(crc16 == receivedCrc16)
116 {
117 qDebug() << "[EXECD] Command executed successfully";
118 emit ping();
119 }
120// else
121// qWarning() << "[EXECD] CRC16 mismatch";
122 emit metrics(raw, data.size(), false);
123 }

134 QByteArray ExecdSocket::wrap(const QString& string) noexcept
135 {
136 // prevents UID overflow
137 if(m_message_uid >= 9999)
138 m_message_uid = 0;
139
140 QString command = ":" + QStringLiteral("%1").arg(++m_message_uid, 4, 10, QLatin1Char('0')) + "|";
141 QString hex_length = QString("%1").arg(string.length(), 2, 16, QLatin1Char('0'));
142 command.append(hex_length + "|" + string + "|");
143 command.append(QStringLiteral("%1").arg(Utils::crc16(Utils::str_data(command), command.length()),
144 4, 16, QLatin1Char('0')));
145 return command.toUtf8();
146 }
