60 void BaseUDPSocket::start(const QString& address)
61 {
62 if(not address.contains(":"))
63 throw std::invalid_argument("Address string must contain semicolon");
64
65 auto spl = address.split(':');
66 if(not address.contains("AnyIPv4") and spl.first().split('.').size() != 4)
67 throw std::invalid_argument("Provided invalid address");
68
69 QHostAddress ip;
70 if(address.contains("AnyIPv4"))
71 ip = QHostAddress(QHostAddress::AnyIPv4);
72 else
73 ip = QHostAddress(spl.first());
74
75 bool uint_success = false;
76 uint16_t p = spl.last().toUInt(&uint_success);
77 if(not uint_success)
78 throw std::invalid_argument("Provided incorrect port");
79
80 qInfo().noquote() << "[UDP] Starting" << name() << "UDP socket on" << address;
81 m_address = ip;
82 m_port = p;
83
84 this->bind(hostAddress(), port());
85 emit metrics("Socket opened on " + addressString(), 0, true);
86 }
