48QVariant ImageModel::data(const QModelIndex& index, int role) const
49{
50 if(not index.isValid())
51 return QVariant();
52
53 switch (role)
54 {
55 case Index: return index.row();
56 case Filename: return QVariant::fromValue(storage[index.row()].filename);
57 case Latitude: return QVariant::fromValue(storage[index.row()].meta.latitude);
58 case Longitude: return QVariant::fromValue(storage[index.row()].meta.longitude);
59 case DX: return QVariant::fromValue(storage[index.row()].meta.dx);
60 case DY: return QVariant::fromValue(storage[index.row()].meta.dy);
61 case X0: return QVariant::fromValue(storage[index.row()].meta.x0);
62 case Y0: return QVariant::fromValue(storage[index.row()].meta.y0);
63 case Angle: return QVariant::fromValue(storage[index.row()].meta.angle);
64 case DriftAngle: return QVariant::fromValue(storage[index.row()].meta.drift_angle);
65 case LX: return QVariant::fromValue(storage[index.row()].meta.lx);
66 case LY: return QVariant::fromValue(storage[index.row()].meta.ly);
67 case Div: return QVariant::fromValue(storage[index.row()].meta.div);
68 case Velocity: return QVariant::fromValue(storage[index.row()].meta.velocity);
69 case Altitude: return QVariant::fromValue(storage[index.row()].meta.altitude);
70 case KR: return QVariant::fromValue(storage[index.row()].meta.kr);
71 case TimeShift: return QVariant::fromValue(storage[index.row()].meta.time_shift);
72 case TimeDuration: return QVariant::fromValue(storage[index.row()].meta.time_duration);
73 case Mode: return QVariant::fromValue(storage[index.row()].meta.mode);
74 case Crc16: return QVariant::fromValue(QString("0x" + QString::number(storage[index.row()].meta.crc16, 16)));
75 case Valid: return QVariant::fromValue(storage[index.row()].valid);
76 case LOD1FilePath: return QVariant::fromValue(storage[index.row()].path.second);
77 case LOD0FilePath: return QVariant::fromValue(storage[index.row()].path.first);
78 case Transparency: return QVariant::fromValue(storage[index.row()].opacity);
79 case Shown: return QVariant::fromValue(storage[index.row()].shown);
80 case MercatorZoomLevel: return QVariant::fromValue(storage[index.row()].mercator_zoom_level);
81 case MarkedForExport: return QVariant::fromValue(storage[index.row()].marked_for_export);
82
83 default: return "Error reading from model";
84 }
85}

87bool ImageModel::setData(const QModelIndex& index, const QVariant& value, int role)
88{
89 if (data(index, role) != value)
90 {
91 switch (role)

93 case Index: return false;

96 case Longitude: storage[index.row()].meta.longitude = value.toDouble(); break;
97 case DX: storage[index.row()].meta.dx = value.toFloat(); break;
98 case DY: storage[index.row()].meta.dy = value.toFloat(); break;
99 case X0: storage[index.row()].meta.x0 = value.toFloat(); break;
100 case Y0: storage[index.row()].meta.y0 = value.toFloat(); break;
101 case Angle: storage[index.row()].meta.angle = value.toFloat(); break;
102 case DriftAngle: storage[index.row()].meta.drift_angle = value.toFloat(); break;
103 case LX: storage[index.row()].meta.lx = value.toFloat(); break;
104 case LY: storage[index.row()].meta.ly = value.toFloat(); break;
105 case Div: storage[index.row()].meta.div = value.toFloat(); break;
106 case Velocity: storage[index.row()].meta.velocity = value.toFloat(); break;
107 case Altitude: storage[index.row()].meta.altitude = value.toFloat(); break;
108 case KR: storage[index.row()].meta.kr = value.toFloat(); break;
109 case TimeShift: storage[index.row()].meta.time_shift = value.toFloat(); break;
110 case TimeDuration: storage[index.row()].meta.time_duration = value.toFloat(); break;
111 case Mode: storage[index.row()].meta.mode = value.toFloat(); break;
112 case Crc16: storage[index.row()].meta.crc16 = value.toUInt(); break;
113 case Valid: storage[index.row()].valid = value.toBool(); break;
114 case LOD1FilePath: storage[index.row()].path.second = value.toString(); break;
115 case LOD0FilePath: storage[index.row()].path.first = value.toString(); break;
116 case Transparency: storage[index.row()].opacity = value.toFloat(); break;
117 case Shown: storage[index.row()].shown = value.toBool(); break;
118 case MercatorZoomLevel: storage[index.row()].mercator_zoom_level = value.toDouble(); break;
119 case MarkedForExport: storage[index.row()].marked_for_export = value.toBool(); break;
120
121 default: return false;
122 }
123 emit dataChanged(index, index, {role});
124 return true;
125 }

191bool ImageModel::exportSelectedImages(const QString& target) noexcept
192{
193 QList<QString> res;
194 for(const auto& image : qAsConst(storage))
195 {
196 if(image.marked_for_export)
197 res.push_back(image.path.first);
198 }
199
200 if(not res.empty()) {
201 qInfo() << "[IMAGE] Marked" << res.size() << "images for export. Exporting...";
202 emit markedForExport(res, target);
203 return true;
204 }
205 else {
206 qWarning() << "[IMAGE] No images marked for export";
207 return false;
208 }
209}
