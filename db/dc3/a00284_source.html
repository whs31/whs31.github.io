105 void TelemetrySocket::processTelemetry(QByteArray data)
106 {
107 uint32_t marker = *(uint32_t*) data.data();
108 if(marker != RECV_MARKER_LITTLE)
109 {
110 qWarning() << "[TELEMETRY] Triggered first marker mismatch";
111 return;
112 }
113
114 QDataStream stream(&data, ReadOnly);
115 stream.setByteOrder(QDataStream::BigEndian);
116 Datagrams::TelemetryDatagram received;
117 stream >> received;
118
119 if(received.marker != RECV_MARKER_BIG)
120 {
121 qWarning() << "[TELEMETRY] Triggered second marker mismatch";
122 return;
123 }
124
125 const float MS_TO_KMH = 3.6;
126
127 received.latitude = Utils::rad2deg(received.latitude);
128 received.longitude = Utils::rad2deg(received.longitude);
129 received.velocity_course = received.velocity_course * MS_TO_KMH;
130 received.velocity_east = received.velocity_east * MS_TO_KMH;
131 received.velocity_north = received.velocity_north * MS_TO_KMH;
132 received.velocity_vertical = received.velocity_vertical * MS_TO_KMH;
133 received.pitch = Utils::rad2deg(received.pitch);
134 received.roll = Utils::rad2deg(received.roll);
135 received.yaw = Utils::rad2deg(received.yaw);
136 received.course = received.yaw;
137 received.satellites = received.satellites >> 1;
138
139 output->setDatagram(received);
140
141 uint16_t crc = checkCRC() ? Utils::crc16_ccitt((const char*) &received, sizeof(Datagrams::TelemetryDatagram) - sizeof(uint16_t))
142 : received.crc16;
143 if(crc != received.crc16)
144 qWarning().noquote().nospace() << "[TELSOCK] Checksum mismatch [" << crc << " : " << received.crc16 << "]";
145
146 emit metrics("0x" + QString::number(received.marker, 16) + " " + QString::number(received.version) + " "
147 + QString::number(received.latitude, 'f', 7) + " " +
148 QString::number(received.longitude, 'f', 7) + " "
149 + QString::number(received.altitude, 'f', 2) + " " +
150 QString::number(received.velocity_course, 'f', 1) + " "
151 + QString::number(received.velocity_east, 'f', 1) + " " +
152 QString::number(received.velocity_north, 'f', 1) + " "
153 + QString::number(received.velocity_vertical, 'f', 1) + " " +
154 QString::number(received.pitch, 'f', 2) + " "
155 + QString::number(received.roll, 'f', 2) + " " + QString::number(received.yaw, 'f', 2) + " "
156 + QString::number(received.course, 'f', 2) + " " + QString::number(received.time) + " " +
157 QString::number(received.satellites)
158 + " 0x" + QString::number(received.crc16, 16), sizeof(received), false);
159 emit ping();
160 }

162 void TelemetrySocket::requestTelemetry()
163 {
164 QByteArray buffer;
165 QDataStream stream(&buffer, QIODevice::WriteOnly);
166 stream.setByteOrder(QDataStream::BigEndian);
167 stream.setFloatingPointPrecision(QDataStream::DoublePrecision);
168
169 Datagrams::TelemetryRequest request = { MARKER, 0x01, (uint16_t)(recv_port), (uint32_t)(this->frequency() * 1'000), 0 };
170 uint16_t crc = Utils::crc16_ccitt((const char*)&request, sizeof(Datagrams::TelemetryRequest) - sizeof(uint16_t));
171 request.crc16 = crc;
172
173 stream << request;
174
175 m_requestSocket->writeDatagram(buffer, QHostAddress(request_addr), request_port);
176 emit metrics("0x" + QString::number(request.marker, 16) + " " + QString::number(request.init_flag) + " "
177 + QString::number(request.port) + " " + QString::number(request.interval_ms)
178 + " 0x" + QString::number(request.crc16, 16), sizeof(request), true);
179 }
