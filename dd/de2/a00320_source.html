20 initialize_qrc_file_within_namespace();
21 const QString CONFIG_PATH = QCoreApplication::applicationDirPath() + "/config";
22 QDir config_directory(CONFIG_PATH);
23 if(not config_directory.exists())
24 config_directory.mkpath(CONFIG_PATH);
25 QFile file(CONFIG_PATH + "/sdk-config.json");
26 if(not file.exists())
27 QFile::copy(":/quasarsdk/rc/sdk-config.json", CONFIG_PATH + "/sdk-config.json");
28
29 if(file.open(QIODevice::ReadOnly | QIODevice::Text))
30 {
31 QByteArray jsonData = file.readAll();
32 file.close();
33
34 QJsonDocument doc = QJsonDocument::fromJson(jsonData);
35 if(not doc.isNull() and doc.isObject())
36 {
37 QJsonObject json = doc.object();
38 for(auto it = json.begin(); it != json.end(); ++it)
39 m_values.insert(it.key(), it.value().toVariant());
40 }
41 }
42 }
