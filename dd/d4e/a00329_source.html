171 void QuasarAPI::setCurrentNetworkDelay(float o)
172 {
173 if(m_currentNetworkDelay == o)
174 return;
175 m_currentNetworkDelay = o;
176 emit currentNetworkDelayChanged();
177
178 if(m_currentNetworkDelay > Config::get()->value<float>("NETWORK_DELAY_THRESHOLD_DISCONNECT"))
179 setConnected(false);
180 else
181 setConnected(true);
182 }

295 void QuasarAPI::execute(QuasarSDK::Enums::NetworkCommand command) noexcept
296 {
297 switch(command)
298 {
299 case Enums::FormImage:
300 case Enums::SimpleStrip:
301 this->setArgument("--remote", m_remote_address_list[0], Enums::Form);
302 break;
303 case Enums::FocusImage:
304 this->setArgument("--remote", m_remote_address_list[0], Enums::Focus);
305 break;
306 case Enums::ReformImage:
307 this->setArgument("--remote", m_remote_address_list[0], Enums::Reform);
308 break;
309 case Enums::StartStrip:
310 case Enums::StopStrip:
311 this->setArgument("--remote", m_remote_address_list[1], Enums::Form);
312 break;
313 default: break;
314 }
315
316 execdSocket()->execute(command);
317 }
