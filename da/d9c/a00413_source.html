169 void QuasarAPI::setCurrentNetworkDelay(float o)
170 {
171 if(m_currentNetworkDelay == o)
172 return;
173 m_currentNetworkDelay = o;
174 emit currentNetworkDelayChanged();
175
176 if(m_currentNetworkDelay > Config::get()->value<float>("NETWORK_DELAY_THRESHOLD_DISCONNECT"))
177 setConnected(false);
178 else
179 setConnected(true);
180 }

286 void QuasarAPI::execute(QuasarSDK::Enums::NetworkCommand command) noexcept
287 {
288 switch(command)
289 {
290 case Enums::FormImage:
291 case Enums::SimpleStrip:
292 this->setArgument("--remote", m_remote_address_list[0], Enums::Form);
293 break;
294 case Enums::FocusImage:
295 this->setArgument("--remote", m_remote_address_list[0], Enums::Focus);
296 break;
297 case Enums::ReformImage:
298 this->setArgument("--remote", m_remote_address_list[0], Enums::Reform);
299 break;
300 case Enums::StartStrip:
301 case Enums::StopStrip:
302 this->setArgument("--remote", m_remote_address_list[1], Enums::Form);
303 break;
304 default: break;
305 }
306
307 execdSocket()->execute(command);
308 }
