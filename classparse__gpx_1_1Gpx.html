stop = self.SeekOffset(Tshift+Ts);
start = self.SeekOffset(Tshift);

lon = self.lon[start:stop];

#print "Lats:", lat

ay = lat[1] - lat[0]
ax = lon[1] - lon[0]

alphav = np.arccos( (ax ) / np.sqrt(ax**2 + ay**2) )

#print ax, ay

#print "Alpha:", np.rad2deg(alphav)

return alphav + np.pi/2
