108 void ExecdSocket::process(QByteArray data)
109 {
110 QString raw = data.data();
111 QString check_crc = raw;
112 check_crc.chop(5);
113
114 uint16_t crc16 = Utils::crc16(Utils::str_data(check_crc), check_crc.length());
115 uint16_t receivedCrc16 = raw.split("|").last().toUInt(nullptr, 16);
116
117 if(crc16 == receivedCrc16)
118 {
119 qDebug() << "[EXECD] Command executed successfully";
120 emit ping();
121 }
122// else
123// qWarning() << "[EXECD] CRC16 mismatch";
124 emit metrics(raw, data.size(), false);
125 }

136 QByteArray ExecdSocket::wrap(const QString& string) noexcept
137 {
138 // prevents UID overflow
139 if(m_message_uid >= 9999)
140 m_message_uid = 0;
141
142 QString command = ":" + QStringLiteral("%1").arg(++m_message_uid, 4, 10, QLatin1Char('0')) + "|";
143 QString hex_length = QString("%1").arg(string.length(), 2, 16, QLatin1Char('0'));
144 command.append(hex_length + "|" + string + "|");
145 command.append(QStringLiteral("%1").arg(Utils::crc16(Utils::str_data(command), command.length()),
146 4, 16, QLatin1Char('0')));
147 return command.toUtf8();
148 }
