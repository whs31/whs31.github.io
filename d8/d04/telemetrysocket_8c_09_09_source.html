22void TelemetrySocket::start(const QString& address, const QString& recv_address)
23{
24 if(not address.contains(":") or address.split(":").size() != 2) {
25 qCritical() << "[TELEMETRY] Provided incorrect request address";
26 return;
27 }
28
29 req_addr = address.split(":").first();
30 req_port = address.split(":").last().toUInt();
31 recv_port = recv_address.split(":").last().toUInt();
32
33 this->connect(recv_address);
34 this->requestTelemetry();
35 m_updateTimer->start((int)(30 * 1'000));
36 qDebug().noquote() << "[TELEMETRY] Started reading at frequency of" << 1 / frequency() << "Hz";
37}

46void TelemetrySocket::processTelemetry(QByteArray data)
47{
48 uint32_t marker = *(uint32_t*)data.data();
49 if(marker != RECV_MARKER_LITTLE) {
50 qWarning() << "[TELEMETRY] Triggered first marker mismatch";
51 return;
52 }
53
54 QDataStream stream(&data, ReadOnly);
55 stream.setByteOrder(QDataStream::BigEndian);
56 TelemetryDatagram received;
57 stream >> received;
58
59 if(received.marker != RECV_MARKER_BIG) {
60 qWarning() << "[TELEMETRY] Triggered second marker mismatch";
61 return;
62 }
63
64 output->setLatitude(LPVL::rad2deg(received.latitude));
65 output->setLongitude(LPVL::rad2deg(received.longitude));
66 output->setAltitude(received.altitude);
67 output->setVelocityCourse(received.velocity_course * 3.6);
68 output->setVelocityEast(received.velocity_east * 3.6);
69 output->setVelocityNorth(received.velocity_north * 3.6);
70 output->setVelocityCourse(received.velocity_course * 3.6);
71 output->setPitch(LPVL::rad2deg(received.pitch));
72 output->setRoll(LPVL::rad2deg(received.roll));
73 output->setYaw(LPVL::rad2deg(received.yaw));
74 output->setCourse(LPVL::rad2deg(received.course));
75 output->setTime(received.time);
76 output->setSatellites(received.satellites >> 1);
77
78 uint16_t crc = CRC_CHECK ? LPVL::crc16_ccitt((const char*)&received, sizeof(Networking::TelemetryDatagram) - sizeof(uint16_t))
79 : received.crc16;
80 if(crc != received.crc16)
81 qWarning().noquote().nospace() << "[TELSOCK] Checksum mismatch [" << crc << " : " << received.crc16 << "]";
82
83 emit socketMetrics("0x" + QString::number(received.marker, 16) + " " + QString::number(received.version) + " "
84 + QString::number(received.latitude, 'f', 7) + " " + QString::number(received.longitude, 'f', 7) + " "
85 + QString::number(received.altitude, 'f', 2) + " " + QString::number(received.velocity_course, 'f', 1) + " "
86 + QString::number(received.velocity_east, 'f', 1) + " " + QString::number(received.velocity_north, 'f', 1) + " "
87 + QString::number(received.velocity_vertical, 'f', 1) + " " + QString::number(received.pitch, 'f', 2) + " "
88 + QString::number(received.roll, 'f', 2) + " " + QString::number(received.yaw, 'f', 2) + " "
89 + QString::number(received.course, 'f', 2) + " " + QString::number(received.time) + " " + QString::number(received.satellites)
90 + " 0x" + QString::number(received.crc16, 16), sizeof(received), false);
91 emit ping();
92}

94void TelemetrySocket::requestTelemetry()
95{
96 QByteArray buffer;
97 QDataStream stream(&buffer, QIODevice::WriteOnly);
98 stream.setByteOrder(QDataStream::BigEndian);
99 stream.setFloatingPointPrecision(QDataStream::DoublePrecision);
100
101 TelemetryRequest request = { MARKER, 0x01, (uint16_t)(recv_port), (uint32_t)(this->frequency() * 1'000), 0 };
102 uint16_t crc = LPVL::crc16_ccitt((const char*)&request, sizeof(TelemetryRequest) - sizeof(uint16_t));
103 request.crc16 = crc;
104
105 stream << request;
106
107 m_requestsock->writeDatagram(buffer, QHostAddress(req_addr), req_port);
108 emit socketMetrics("0x" + QString::number(request.marker, 16) + " " + QString::number(request.init_flag) + " "
109 + QString::number(request.port) + " " + QString::number(request.interval_ms)
110 + " 0x" + QString::number(request.crc16, 16), sizeof(request), true);
111}
