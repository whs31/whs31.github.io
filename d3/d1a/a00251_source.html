169 void QuasarAPI::setCurrentNetworkDelay(float o)
170 {
171 if(m_currentNetworkDelay == o)
172 return;
173 m_currentNetworkDelay = o;
174 emit currentNetworkDelayChanged();
175
176 if(m_currentNetworkDelay > Config::get()->value<float>("NETWORK_DELAY_THRESHOLD_DISCONNECT"))
177 setConnected(false);
178 else
179 setConnected(true);
180 }

293 void QuasarAPI::execute(QuasarSDK::Enums::NetworkCommand command) noexcept
294 {
295 switch(command)
296 {
297 case Enums::FormImage:
298 case Enums::SimpleStrip:
299 this->setArgument("--remote", m_remote_address_list[0], Enums::Form);
300 break;
301 case Enums::FocusImage:
302 this->setArgument("--remote", m_remote_address_list[0], Enums::Focus);
303 break;
304 case Enums::ReformImage:
305 this->setArgument("--remote", m_remote_address_list[0], Enums::Reform);
306 break;
307 case Enums::StartStrip:
308 case Enums::StopStrip:
309 this->setArgument("--remote", m_remote_address_list[1], Enums::Form);
310 break;
311 default: break;
312 }
313
314 execdSocket()->execute(command);
315 }
