98QString ExecdSocket::wrap(const QString& string)

100 // prevents UID overflow

104 QString command = ":" + QStringLiteral("%1").arg(++message_uid, 4, 10, QLatin1Char('0')) + "|";

115void ExecdSocket::processResult(QByteArray data)
116{
117 QString raw = data.data();
118 QString check_crc = raw;
119 check_crc.chop(5);
120
121 //crc16
122 uint16_t crc16 = LPVL::crc16(LPVL::str_data(check_crc), check_crc.length());
123 uint16_t receivedCrc16 = raw.split("|").last().toUInt(nullptr, 16);
124
125 if(crc16 == receivedCrc16) {
126 qDebug() << "[EXECD] Command executed successfully";
127 emit ping();
128 }
129 else
130 qWarning() << "[EXECD] CRC16 mismatch";
