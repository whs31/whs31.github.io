105 {
106 QString raw = data.data();
107 QString check_crc = raw;
108 check_crc.chop(5);
109
110 uint16_t crc16 = SDK::Crypto::crc16(SDK::Crypto::str_data(check_crc), check_crc.length());
111 uint16_t receivedCrc16 = raw.split("|").last().toUInt(nullptr, 16);

113 if(crc16 == receivedCrc16)
