120 void Network::executeCommand(Networking::Enums::NetworkCommand command) const noexcept
121 {
122 switch(command)
123 {
124 case Enums::FormImage:
125 this->setArgument("--remote", stringifyIP(CONFIG(localIP), CONFIG(tcpLFSPort)), Enums::Form);
126 break;
127 case Enums::SimpleStrip:
128 this->setArgument("--remote", stringifyIP(CONFIG(localIP), CONFIG(tcpLFSPort)), Enums::Form);
129 break;
130 case Enums::FocusImage:
131 this->setArgument("--remote", stringifyIP(CONFIG(localIP), CONFIG(tcpLFSPort)), Enums::Focus);
132 break;
133 case Enums::ReformImage:
134 this->setArgument("--remote", stringifyIP(CONFIG(localIP), CONFIG(tcpLFSPort)), Enums::Reform);
135 break;
136 case Enums::StartStrip:
137 if(ICFG<bool>("NETWORK_MERGE_FSEND_SOCKETS"))
138 this->setArgument("--remote", stringifyIP(CONFIG(localIP), CONFIG(tcpLFSPort)), Enums::Form);
139 else
140 this->setArgument("--remote", stringifyIP(CONFIG(localIP), CONFIG(udpLFSPort)), Enums::Form);
141 break;
142 case Enums::StopStrip:
143 if(ICFG<bool>("NETWORK_MERGE_FSEND_SOCKETS"))
144 this->setArgument("--remote", stringifyIP(CONFIG(localIP), CONFIG(tcpLFSPort)), Enums::Form);
145 else
146 this->setArgument("--remote", stringifyIP(CONFIG(localIP), CONFIG(udpLFSPort)), Enums::Form);
147 break;
148
149 default: break;
150 }
151
152 execdSocket->executeCommand(command);
153 }
